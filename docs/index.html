npm-grep

  • Is npm search too slow for you?
  • Don't mind slightly out of date information?
  • Want to be able to "drill-down" to get more details?

This little module will allow you to search the contents of your local npm cache, usually saved in ~/.npm/-/all/.cache.json and which has over 8000 entries (as of March 2012).

 
(function() {

// var sys = require('util'); // only 1 dependency: sys.puts!

var home=process.env['HOME']

var ncache = require(home + '/.npm/-/all/.cache.json'); typeof ncache;

ngrep(/regex/) will search for all modules in your npm cache containing the work "regex". Can also filter by a function: ngrep(function(d) { return d.length > 240; }) will show all entries where the name + description is more than 240 characters long.

 
function ngrep(re) { 
  var fn = "function" === typeof re ? re : function(it){ return it.match(re)}; 
  return ngrep.last=descr.filter(fn); 
}
